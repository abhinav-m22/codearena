2 // Find two numbers in the array that add up to the target
3 
4 function twoSum(nums, target) {
5   const map = new Map();
6   
7   for (let i = 0; i < nums.length; i++) {
8     const complement = target - nums[i];
9     
10     if (map.has(complement)) {
11       return [map.get(complement), i];
12     }
13     
14     map.set(nums[i], i);
15   }
16   
17   return []; // No solution found
18 }
19 
20 // Test cases
21 console.log(twoSum([2, 7, 11, 15], 9)); // Output: [0, 1]
22 console.log(twoSum([3, 2, 4], 6));      // Output: [1, 2]
